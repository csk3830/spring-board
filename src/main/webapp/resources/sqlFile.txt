-- 24. 10. 29.
create table board(
bno bigint auto_increment,
title varchar(500) not null,
writer varchar(500) not null,
content text,
is_del varchar(5) default 'N',
reg_date datetime default now(),
read_count int default 0,
primary key(bno)
);

-- 24. 10. 31.
create table comment(
cno bigint auto_increment,
bno bigint not null ,
writer varchar(500) not null, 
content text, 
reg_date datetime default now(),
primary key(cno));

-- 24. 11. 01.
create table file(
uuid varchar(256) not null,
save_dir varchar(256) not null,
file_name varchar(256) not null,
file_type tinyint(1) default 0,
bno bigint,
file_size  bigint,
reg_date datetime default now(),
primary key(uuid)
);

-- 24. 11. 04.
alter TABLE board
add cmt_qty int default 0,
add has_file int default 0;

-- 댓글 수와 파일 수 업데이트
update board b set cmt_qty = (
SELECT COUNT(bno) FROM comment WHERE bno = b.bno);

update board b set has_file = (
SELECT COUNT(bno) FROM file WHERE bno = b.bno);
